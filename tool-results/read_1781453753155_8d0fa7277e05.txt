    15→// Helper: execute a single SQL statement, ignoring "already exists" type errors
    16→async function execIgnore(db: D1Database, sql: string): Promise<{ sql: string; ok: boolean; error?: string }> {
    17→  try {
    18→    await db.exec(sql);
    19→    return { sql: sql.substring(0, 80), ok: true };
    20→  } catch (e: any) {
    21→    const msg = e?.message || String(e);
    22→    // Ignore errors for already-existing columns/tables/indexes
    23→    if (
    24→      msg.includes('already exists') ||
    25→      msg.includes('duplicate column name') ||
    26→      msg.includes('UNIQUE constraint failed')
    27→    ) {
    28→      return { sql: sql.substring(0, 80), ok: true, error: `ignored: ${msg}` };
    29→    }
    30→    return { sql: sql.substring(0, 80), ok: false, error: msg };
    31→  }
    32→}
