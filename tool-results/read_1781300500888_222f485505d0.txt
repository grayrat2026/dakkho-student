  1568тЖТ  1568тЖТasync function checkDailyEmailRateLimit(db: D1Database, email: string): Promise<{ allowed: boolean; count: number; limit: number }> {
  1569тЖТ  1569тЖТ  try {
  1570тЖТ  1570тЖТ    // Count OTPs created for this email in the last 24 hours (both purposes)
  1571тЖТ  1571тЖТ    // NOTE: created_at is stored in ISO 8601 format, so we validate in JavaScript
  1572тЖТ  1572тЖТ    // to avoid SQLite datetime format mismatch issues.
  1573тЖТ  1573тЖТ    const now = new Date();
  1574тЖТ  1574тЖТ    const twentyFourHoursAgo = new Date(now.getTime() - 24 * 60 * 60 * 1000).toISOString();
  1575тЖТ  1575тЖТ    const result = await db.prepare(
  1576тЖТ  1576тЖТ      'SELECT created_at FROM password_reset_otps WHERE email = ?'
  1577тЖТ  1577тЖТ    ).bind(email).all<{ created_at: string | null }>();
  1578тЖТ  1578тЖТ
  1579тЖТ  1579тЖТ    // Filter in JavaScript to handle ISO 8601 dates correctly
  1580тЖТ  1580тЖТ    const recentCount = result.results.filter(row => {
  1581тЖТ  1581тЖТ      if (!row.created_at) return false;
  1582тЖТ  1582тЖТ      try {
  1583тЖТ  1583тЖТ        return new Date(row.created_at) >= new Date(twentyFourHoursAgo);
  1584тЖТ  1584тЖТ      } catch {
  1585тЖТ  1585тЖТ        return false;
  1586тЖТ  1586тЖТ      }
  1587тЖТ  1587тЖТ    }).length;
  1588тЖТ  1588тЖТ
  1589тЖТ  1589тЖТ    return { allowed: recentCount < EMAIL_OTP_DAILY_LIMIT, count: recentCount, limit: EMAIL_OTP_DAILY_LIMIT };
  1590тЖТ  1590тЖТ  } catch {
  1591тЖТ  1591тЖТ    // If the check fails, allow the request (fail open)
  1592тЖТ  1592тЖТ    return { allowed: true, count: 0, limit: EMAIL_OTP_DAILY_LIMIT };
  1593тЖТ  1593тЖТ  }
  1594тЖТ  1594тЖТ}
